\section{Introduction}\label{introduction}

\begin{frame}{Introduction}
\hypertarget{sec:intro}{}

\small\normalfont
\begin{columns}[T,onlytextwidth]
  \begin{column}{0.58\textwidth}
  \begin{itemize}
    \item<1-> Modern data centers consume substantial electricity for both computing and cooling.
    \vspace{0.5em}
    \item<2-> Operators must balance computational demand, operating cost, and environmental impact.
    \vspace{0.5em}
    \item<3-> This study analyzes hourly workload, electricity prices, server parameters, and adsorption cooling system operating states.
    \vspace{0.5em}
    \item<4-> The goal is to show how data analytics and machine learning can support smarter data center planning.
  \end{itemize}
  \end{column}

  \begin{column}{0.42\textwidth}
    \only<1-2>{\centering\includegraphics[width=\linewidth]{Data Center images/Figure_1_Hourly_Workload.svg}\par}
    \only<3->{\centering\includegraphics[width=\linewidth]{Data Center images/Figure_3_Power_Breakdown.svg}\par}
  \end{column}
\end{columns}
\end{frame}

\begin{frame}{Motivation}
\phantomsection\label{motivation}
\begin{itemize}
\tightlist
\item
  Data center operation depends on three strongly coupled factors:

  \begin{itemize}
  \tightlist
  \item
    workload demand,
  \item
    electricity price,
  \item
    cooling-system performance.
  \end{itemize}
\end{itemize}

\vspace{0.8em}

\begin{itemize}
\tightlist
\item
  Peak workload hours do not always align with high-price hours.
\item
  Cooling efficiency is strongly nonlinear and depends on operating
  temperatures.
\item
  These effects create opportunities for AI-driven and cost-aware
  control.
\end{itemize}
\end{frame}

\begin{frame}{Objectives}
\phantomsection\label{objectives}
\begin{itemize}
\tightlist
\item
  Characterize daily workload and electricity-price behavior.
\item
  Estimate derived server and facility power behavior.
\item
  Evaluate hourly electricity cost and carbon emissions.
\item
  Analyze the adsorption cooling system performance map.
\item
  Build an AI surrogate model for cooling efficiency and cooling
  capacity.
\end{itemize}
\end{frame}

\section{Dataset and Methodology}\label{dataset-and-methodology}

\begin{frame}{Dataset and Methodology}
\hypertarget{sec:data}{}
\end{frame}

\begin{frame}{Dataset Overview}
\phantomsection\label{dataset-overview}
The dataset contains the following structured sheets:

\begin{itemize}
\tightlist
\item
  Workload Curves
\item
  Electricity Price Curves
\item
  Server Parameters
\item
  ADS Operating States
\item
  Planning Parameters
\item
  Carbon Emission
\item
  Cooling-related Equipment
\item
  Physical Properties
\end{itemize}

\vspace{0.8em}

The analysis focuses on workload behavior, power demand, cost,
emissions, and cooling-system performance.
\end{frame}

\begin{frame}{Derived Operational Quantities}
\phantomsection\label{derived-operational-quantities}
From the hourly workload and engineering parameters, the following
operational metrics are estimated:

\begin{itemize}
\tightlist
\item
  required server count,
\item
  server operating frequency,
\item
  IT power demand,
\item
  cooling power demand,
\item
  total facility power,
\item
  electricity cost,
\item
  carbon emissions.
\end{itemize}
\end{frame}

\begin{frame}{AI Model}
\phantomsection\label{ai-model}
\begin{itemize}
  \item Inputs: generator temperature and condenser temperature
  \item Outputs: coefficient of performance (COP) and cooling capacity
  \item Model: Random Forest multi-output regression
\end{itemize}

\vspace{0.8em}

The ADS operating-state table is well suited for supervised learning
because it contains a dense grid of operating points.
\end{frame}

\begin{frame}{Key Equations}
\phantomsection\label{key-equations}
\small

{[} \text{Electricity Cost} = \text{Energy Consumption}
\times \text{Electricity Price} {]}

{[} \text{Carbon Emissions} = \text{Energy Consumption}
\times \text{Carbon Intensity} {]}

{[} (\widehat{\text{COP}}, \widehat{q}\^{}\{cap\}) = f(T\^{}\{gen\},
T\^{}\{cond\}) {]}

{[} R\^{}2 = 1 -
\frac{\sum_i (y_i - \hat y_i)^2}{\sum_i (y_i - \bar y)^2} {]}
\end{frame}

\item
  This supports the value of predictive and cost-aware operation.
\end{itemize}

    \centering
    \includegraphics[width=\linewidth]{Data Center images/Figure_4_Cost_and_Carbon.svg}

\begin{itemize}

  Higher workload increases both IT power and cooling demand.
\item

  variation.
\item
  Cost and carbon should therefore be evaluated jointly.
\end{itemize}
\end{frame}

\section{Conclusion and Future Work}\label{conclusion-and-future-work}

\begin{frame}{Conclusion and Future Work}
\hypertarget{sec:conclusion}{}
\end{frame}

\begin{frame}{Conclusion}
\phantomsection\label{conclusion}
\begin{itemize}
\tightlist
\item
  Workload and electricity-price profiles reveal meaningful scheduling
  opportunities.
\item
  Facility power, cost, and carbon emissions vary substantially over the
  day.
\item
  ADS performance is highly nonlinear.
\item
  A Random Forest surrogate accurately predicts both COP and cooling
  capacity.
\end{itemize}

\vspace{0.8em}

Overall, AI-driven operational analytics can help data centers become
more energy-efficient, cost-aware, and environmentally responsible.
\end{frame}

\begin{frame}{Future Work}
\phantomsection\label{future-work}
\begin{itemize}
\tightlist
\item
  workload forecasting using time-series methods,
\item
  cost-aware batch scheduling,
\item
  multi-objective optimization of energy, cost, and emissions,
\item
  reinforcement learning for adaptive control,
\item
  integration into digital twins and real-time decision support.
\end{itemize}
\end{frame}
